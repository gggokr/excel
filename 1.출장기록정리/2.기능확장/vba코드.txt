질문: 이름 열에는 다양한 사람의 이름이 있다. 이름을 기준으로 데이터를 분리하고, 추출한 이름으로 새로운 시트를 만든다. 새롭게 만든 시트에는 해당하는 데이터를 넣어야 한다. 이해했나?
엑셀VBA를 만들어줘.

추가질문: For Each 컨트롤 변수는 Variant나 Object 이어야 한다는 컴파일 오류가 발생했다.


Sub SplitDataByName()
    Dim ws As Worksheet
    Dim newWs As Worksheet
    Dim lastRow As Long
    Dim nameCol As Range
    Dim cell As Range
    Dim nameDict As Object
    Dim nameValue As Variant ' ← 여기 수정
    Dim headerRow As Range
    Dim dataRange As Range

    Set ws = ThisWorkbook.Sheets("Sheet1") ' 원본 시트 이름
    Set nameDict = CreateObject("Scripting.Dictionary")

    ' 이름 열이 있는 전체 범위 (예: D열이 이름 열이면 D2:D마지막행)
    lastRow = ws.Cells(ws.Rows.Count, "D").End(xlUp).Row
    Set nameCol = ws.Range("D2:D" & lastRow) ' "이름" 열, 제목행 제외

    ' 제목행 저장
    Set headerRow = ws.Rows(1)

    ' 중복 없이 이름 수집
    For Each cell In nameCol
        If Trim(cell.Value) <> "" Then
            If Not nameDict.exists(Trim(cell.Value)) Then
                nameDict.Add Trim(cell.Value), True
            End If
        End If
    Next cell

    ' 이름별로 시트 생성 및 데이터 복사
    For Each nameValue In nameDict.Keys
        On Error Resume Next
        Application.DisplayAlerts = False
        Worksheets(CStr(nameValue)).Delete ' 기존 시트 삭제
        Application.DisplayAlerts = True
        On Error GoTo 0

        Set newWs = Worksheets.Add(After:=Worksheets(Worksheets.Count))
        newWs.Name = Left(CStr(nameValue), 31) ' 시트명은 31자 제한

        ' 제목행 복사
        headerRow.Copy Destination:=newWs.Range("A1")

        ' 해당 이름의 데이터 복사
        Dim i As Long, pasteRow As Long
        pasteRow = 2
        For i = 2 To lastRow
            If Trim(ws.Cells(i, "D").Value) = nameValue Then
                ws.Rows(i).Copy Destination:=newWs.Rows(pasteRow)
                pasteRow = pasteRow + 1
            End If
        Next i
    Next nameValue

    MsgBox "이름별 시트 생성 완료!"
End Sub


질문: 이번에는 다른 코드를 만들어줘.
출발일을 기준으로 월별 데이터를 분리하고 싶다.
데이터 형식은 일반이다.

추가질문: For Each 컨트롤 변수는 Variant나 Object 이어야 한다는 컴파일 오류가 발생했다.


Sub SplitDataByMonth()
    Dim ws As Worksheet
    Dim newWs As Worksheet
    Dim lastRow As Long
    Dim dateColIndex As Long
    Dim headerRow As Range
    Dim i As Long
    Dim dateValue As String
    Dim monthStr As String
    Dim cell As Range
    Dim monthDict As Object
    Dim dataRow As Range
    Dim key As Variant ' ← 수정된 부분

    Set ws = ThisWorkbook.Sheets("Sheet1")
    Set monthDict = CreateObject("Scripting.Dictionary")
    
    ' 제목행 범위
    Set headerRow = ws.Rows(1)
    
    ' "출발일" 열 찾기
    For Each cell In headerRow.Cells
        If Trim(cell.Value) = "출발일" Then
            dateColIndex = cell.Column
            Exit For
        End If
    Next cell
    
    If dateColIndex = 0 Then
        MsgBox "출발일 열을 찾을 수 없습니다."
        Exit Sub
    End If
    
    ' 마지막 행 찾기
    lastRow = ws.Cells(ws.Rows.Count, dateColIndex).End(xlUp).Row
    
    ' 월별로 데이터 수집
    For i = 2 To lastRow
        dateValue = Trim(ws.Cells(i, dateColIndex).Value)
        
        If IsDate(dateValue) Then
            monthStr = Format(CDate(dateValue), "mm") & "월"
            
            If Not monthDict.exists(monthStr) Then
                monthDict.Add monthStr, New Collection
            End If
            
            monthDict(monthStr).Add i ' 행 번호 저장
        End If
    Next i
    
    ' 각 월별 시트 생성 및 데이터 복사
    For Each key In monthDict.Keys
        On Error Resume Next
        Application.DisplayAlerts = False
        Worksheets(CStr(key)).Delete ' 기존 시트 삭제
        Application.DisplayAlerts = True
        On Error GoTo 0
        
        Set newWs = Worksheets.Add(After:=Worksheets(Worksheets.Count))
        newWs.Name = key
        
        ' 제목행 복사
        headerRow.Copy Destination:=newWs.Range("A1")
        
        ' 해당 월에 해당하는 행 복사
        Dim r As Variant
        For Each r In monthDict(key)
            ws.Rows(r).Copy Destination:=newWs.Cells(newWs.Cells(Rows.Count, 1).End(xlUp).Row + 1, 1)
        Next r
    Next key
    
    MsgBox "월별 시트 생성 완료!"
End Sub
